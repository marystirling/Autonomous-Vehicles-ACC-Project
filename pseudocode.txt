function handler(V_0, V_S, V_C, V_R, T_g)
old_time = sys.time()
while sys.time() >= old_time + 0.1:
	if (V_S != V_C): // maybe floor/ceil these? round to even?
		accel(V_0, V_S, V_C, V_R, T_g);
		old_time = sys.time()
